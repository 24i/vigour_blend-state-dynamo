```javascript
state.inject(require('blend-state-dynamo'))

state.set({
  db: {
    id: AMAZON_ID,
    secret: AMAZON_SECRET,
    table: testTable
  },
  on: {
    error (err) {
      console.log(err)
    }
  }
})
state.db.hasTable.is(true).then(() => {
  t.ok(true, 'hasTable')
  t.end()
})
```
